// Simple target: 295 days from now at 6 AM IST let targetDate; const savedTarget = localStorage.getItem('wifeTargetDate'); if (savedTarget) { targetDate = new Date(savedTarget); console.log('Loaded saved target:', targetDate); } else { // Simple approach: 295 days from current date const now = new Date(); targetDate = new Date(now); targetDate.setDate(now.getDate() + 295); targetDate.setHours(6, 0, 0, 0); // Set to 6 AM localStorage.setItem('wifeTargetDate', targetDate.toISOString()); console.log('Created new target:', targetDate); } function updateCountdown() { const now = new Date(); const timeDiff = targetDate - now; // If target reached if (timeDiff <= 0) { document.getElementById('countdown').textContent = '0'; document.getElementById('days').textContent = '0'; document.getElementById('hours').textContent = '0'; document.getElementById('weeks').textContent = '0'; document.getElementById('daysCompleted').textContent = '295'; document.querySelector('.progress-fill').style.setProperty('--progress-width', '100%'); document.querySelector('.status').textContent = '🎊 SHE QUIT HER JOB! IT\'S HAPPENING NOW!'; document.querySelector('.status').style.background = 'rgba(16, 185, 129, 0.4)'; document.querySelector('.status').style.borderColor = 'rgba(16, 185, 129, 0.6)'; return; } // Calculate total hours remaining (for display) const totalHours = Math.ceil(timeDiff / (1000 * 60 * 60)); // Calculate days remaining (rounded up) const daysLeft = Math.ceil(totalHours / 24); // Update displays document.getElementById('countdown').textContent = daysLeft; document.getElementById('days').textContent = daysLeft; document.getElementById('hours').textContent = totalHours; const weeksLeft = Math.ceil(daysLeft / 7); document.getElementById('weeks').textContent = weeksLeft; const daysCompleted = 295 - daysLeft; document.getElementById('daysCompleted').textContent = daysCompleted; // Update progress bar const progressPercent = ((295 - daysLeft) / 295) * 100; document.querySelector('.progress-fill').style.setProperty('--progress-width', progressPercent + '%'); // Fun effects based on time left const hoursLeft = totalHours; if (hoursLeft <= 168 && hoursLeft > 0) { document.querySelector('.status').textContent = '🔥 ONLY ' + daysLeft + ' DAYS (' + hoursLeft + ' HOURS) LEFT!'; document.querySelector('.status').style.background = 'rgba(236, 72, 153, 0.4)'; document.querySelector('.status').style.borderColor = 'rgba(236, 72, 153, 0.7)'; } else if (hoursLeft <= 72) { document.querySelector('.status').textContent = '⚡ JUST ' + daysLeft + ' DAYS REMAINING!'; document.querySelector('.status').style.background = 'rgba(99, 102, 241, 0.4)'; document.querySelector('.status').style.borderColor = 'rgba(99, 102, 241, 0.7)'; } else if (hoursLeft === 0) { document.querySelector('.status').textContent = '🎊 SHE QUIT HER JOB! IT\'S HAPPENING NOW!'; document.querySelector('.status').style.background = 'rgba(16, 185, 129, 0.4)'; document.querySelector('.status').style.borderColor = 'rgba(16, 185, 129, 0.7)'; } else if (hoursLeft <= 24) { document.querySelector('.status').textContent = '🌅 TOMORROW IS THE DAY!'; } else { document.querySelector('.status').textContent = '🎉 She\'s quitting her job soon!'; document.querySelector('.status').style.background = 'rgba(16, 185, 129, 0.2)'; document.querySelector('.status').style.borderColor = 'rgba(16, 185, 129, 0.4)'; } document.getElementById('timezone').textContent = 'IST'; }
